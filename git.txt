git --version   {show version name}

git config --global user.name 
git config --global user.email

code .  {to open code editor}

ls  {list all the files in the directory}

git init {to initialize an existing directory as a Git repository}

git status {to check status}

git add [file name] {to add a file}

git add -A {to add all files at once}   

touch [file name] {to create a file}

git commit -m "message" {to commit all the files}

git commit              {to commit all the files}
    then, press i and enter
        Initial commit   
    then, press Esc and enter   
        :wq

git checkout [file name] {to get a file recovered to the previously committed version}

git checkout -f {to get all files recovered to the previously committed version}

git diff [file name] or git diff {Show changes between working directory and staging area}

git diff --staged {this is another way}

git rm --cached [file name] {to remove only from staging make it untrack}

git rm [file name] {deletes the file}

git status -s       {status in short}

touch .gitignore  {inside this file write the name of files to ignore them from being tracked}
    [file name]     {ignores files from every with same name}
    [/file name]    {ignores files from only root where .gitignore file is}
    [*.ext]         {ignore files with same extension that you entered}
        
mkdir [directory name]  {creates new directory}

cd [directory name]     {changes between directory}

git branch [branch name]  {creates new branch}

git branch      {shows all the branches}

git checkout [branch name]  {switch to branch}

git checkout -b [branch name] {creates as well as switch to the branch}

git merge [branch name]     {merges the current branch with the branch name we include}

git branch -d [branch name] {deletes the branch}
